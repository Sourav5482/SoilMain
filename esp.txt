#include <Arduino.h>
#include <BLEDevice.h>
#include <BLEUtils.h>
#include <BLEServer.h>
#include <BLE2902.h>

#define SOIL_PIN 34

// Must match mobile app config
static const char* DEVICE_NAME = "ESP32_SOIL";
static const char* SERVICE_UUID = "4fafc201-1fb5-459e-8fcc-c5c9c331914b";
static const char* CHARACTERISTIC_UUID = "beb5483e-36e1-4688-b7f5-ea07361b26a8";

BLECharacteristic* sensorChar = nullptr;
bool deviceConnected = false;
unsigned long lastSend = 0;

class MyCharacteristicCallbacks : public BLECharacteristicCallbacks {
  void onWrite(BLECharacteristic* pCharacteristic) override {
    std::string value = pCharacteristic->getValue();
    if (value.empty()) {
      return;
    }

    String command = String(value.c_str());
    command.trim();
    command.toUpperCase();

    if (command == "RESET") {
      Serial.println("RESET command received from app. Restarting...");
      delay(120);
      ESP.restart();
    }
  }
};

class MyServerCallbacks : public BLEServerCallbacks {
  void onConnect(BLEServer* pServer) override {
    deviceConnected = true;
    Serial.println("App connected");
  }

  void onDisconnect(BLEServer* pServer) override {
    deviceConnected = false;
    Serial.println("App disconnected");
    BLEDevice::startAdvertising(); // allow reconnect
  }
};

void setup() {
  Serial.begin(115200);
  analogReadResolution(12); // 0..4095

  BLEDevice::init(DEVICE_NAME);
  BLEServer* pServer = BLEDevice::createServer();
  pServer->setCallbacks(new MyServerCallbacks());

  BLEService* pService = pServer->createService(SERVICE_UUID);

  sensorChar = pService->createCharacteristic(
    CHARACTERISTIC_UUID,
    BLECharacteristic::PROPERTY_READ |
      BLECharacteristic::PROPERTY_NOTIFY |
      BLECharacteristic::PROPERTY_WRITE |
      BLECharacteristic::PROPERTY_WRITE_NR
  );
  sensorChar->addDescriptor(new BLE2902());
  sensorChar->setCallbacks(new MyCharacteristicCallbacks());

  pService->start();

  BLEAdvertising* pAdvertising = BLEDevice::getAdvertising();
  pAdvertising->addServiceUUID(SERVICE_UUID);
  pAdvertising->setScanResponse(true);
  pAdvertising->setMinPreferred(0x06);
  pAdvertising->setMinPreferred(0x12);
  BLEDevice::startAdvertising();

  Serial.println("ESP32 BLE ready: scan and select ESP32_SOIL in app");
}

void loop() {
  if (deviceConnected && millis() - lastSend >= 1000) {
    lastSend = millis();

    // Example soil moisture from analog
    int raw = analogRead(SOIL_PIN);
    int moisturePct = map(raw, 4095, 0, 0, 100);
    moisturePct = constrain(moisturePct, 0, 100);

    // Demo values (replace with your real sensors)
    float temp = 28.0 + (random(-20, 20) / 10.0);  // 26.0..30.0
    float humidity = (float)moisturePct;           // using moisture as humidity
    float n = 12.0 + (random(-10, 10) / 10.0);
    float p = 8.0 + (random(-10, 10) / 10.0);
    float k = 20.0 + (random(-10, 10) / 10.0);

    // Your app now supports payload without lat/lng
    String payload = "{";
    payload += "\"temp\":" + String(temp, 2) + ",";
    payload += "\"humidity\":" + String(humidity, 2) + ",";
    payload += "\"n\":" + String(n, 2) + ",";
    payload += "\"p\":" + String(p, 2) + ",";
    payload += "\"k\":" + String(k, 2);
    payload += "}";

    sensorChar->setValue(payload.c_str());
    sensorChar->notify();

    Serial.println(payload);
  }

  delay(10);
}
